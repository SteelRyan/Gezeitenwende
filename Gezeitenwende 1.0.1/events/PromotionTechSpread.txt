# Guild Based Production
country_event = {
	id = 750001
	title = "EVTNAME750001"
	desc = "EVTDESC750001"
	
	trigger = {
		NOT = {
			has_country_flag = guild_based_production
		}
		guild_based_production = 1
	}
	
	immediate = { set_country_flag = guild_based_production }

	option = { name = "OK" }
}
